        // --- Populate Grid --- 

        function populateGrid(filter = 'all') { 

            grid.innerHTML = ''; 

            const filteredAgencies = filter === 'all' ? agencies : agencies.filter(a => a.category === filter); 

 

            filteredAgencies.forEach(agency => { 

                const card = document.createElement('div');

                grid.appendChild(card); 

            }); 

        } 

 

        // --- Modal Logic --- 

        function openModal(agencyName) { 

            const agency = agencies.find(a => a.name === agencyName); 

            if (agency) { 

                modalTitle.textContent = agency.name; 

                modalSubtitle.textContent = agency.company; 

                modalBody.innerHTML = ` 

                    <div class="space-y-4"> 

                        <div> 

                            <h4 class="font-semibold text-gray-800 mb-1">Naming Category</h4> 

                            <p class="text-gray-600">${agency.category.charAt(0).toUpperCase() + agency.category.slice(1)}</p> 

                        </div> 

                        <div> 

                            <h4 class="font-semibold text-gray-800 mb-1">Name Origin</h4> 

                            <p class="text-gray-600">${agency.origin}</p> 

                        </div> 

                        <div> 

                            <h4 class="font-semibold text-gray-800 mb-1">Key Capabilities</h4> 

                            <p class="text-gray-600">${agency.capabilities}</p> 

                        </div> 

                    </div> 

                `; 

                modal.classList.remove('hidden'); 

                document.body.style.overflow = 'hidden'; 

                 setTimeout(() => { 

                    modal.querySelector('.modal-content').classList.remove('scale-95'); 

                    modal.classList.remove('opacity-0'); 

                }, 10); 

            } 

        } 

 

        function closeModal() { 

            modal.querySelector('.modal-content').classList.add('scale-95'); 

            modal.classList.add('opacity-0'); 

            setTimeout(() => { 

                modal.classList.add('hidden'); 

                document.body.style.overflow = ''; 

            }, 300); 

        } 

 

        closeModalBtn.addEventListener('click', closeModal); 

        modal.addEventListener('click', (e) => { 

            if (e.target === modal) { 

                closeModal(); 

            } 

        }); 

        document.addEventListener('keydown', (e) => { 

            if (e.key === 'Escape' && !modal.classList.contains('hidden')) { 

                closeModal(); 

            } 

        }); 

         

        // --- Filtering Logic --- 

        tags.forEach(tag => { 

            tag.addEventListener('click', () => { 

                const filter = tag.dataset.filter; 

                 

                tags.forEach(t => { 

                    t.classList.remove('bg-indigo-600', 'text-white'); 

                    t.classList.add('bg-white', 'border', 'border-gray-300'); 

                }); 

                 

                tag.classList.add('bg-indigo-600', 'text-white'); 

                tag.classList.remove('bg-white', 'border', 'border-gray-300'); 

 

                populateGrid(filter); 

            }); 

        }); 

 

        // --- Initial Load --- 

        document.addEventListener('DOMContentLoaded', () => {
